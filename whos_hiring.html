where post_type = 'story'
 and  lower(title) like '%ask hn: who is hiring%'
 and  submitter_id = 'whoishiring'

(
  select p.*
  from {{ ref('hn_posts') }} p
  inner join hiring h
  on p.parent_id = h.post_id
)
select *  
from job_posts
limit 50
